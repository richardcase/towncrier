```zig
export fn towncrier_init(rt: ?*const RuntimeCallbacks) callconv(.c) ?*anyopaque {
    if (rt == null) return null;
    const handle = std.heap.c_allocator.create(types.TowncrierHandle) catch return null;
    const alloc = std.heap.c_allocator;
    handle.* = .{
        .allocator = alloc,
        .callbacks = rt.?.*,
        .accounts = std.ArrayList(types.AccountState).init(alloc),
        .action_mutex = .{},

        .snapshot_lock = .{},
        .snapshot = null,
        .db = null,

        alloc.destroy(handle);
        return null;

        handle.accounts.deinit();
        handle.action_queue.deinit();

        alloc.destroy(handle);
        return null;
    };
    return handle;
```

```zig
export fn towncrier_free(tc: ?*anyopaque) callconv(.c) void {
    if (tc) |ptr| {
        const handle: *types.TowncrierHandle = @ptrCast(@alignCast(ptr));
        // Per RESEARCH.md Pitfall 6: stop all poll threads before freeing handle
        towncrier_stop(ptr);
        // Free snapshot if present
        if (handle.snapshot) |snap| {
            snap.string_buf.deinit();
            handle.allocator.free(snap.items);
            handle.allocator.destroy(snap);
        }
        // Free per-account data (token, base_url copies)
        for (handle.accounts.items) |*acct_state| {
            handle.allocator.free(acct_state.account.base_url);
            handle.allocator.free(acct_state.account.token);
            if (acct_state.account.last_modified) |lm| handle.allocator.free(lm);
        }
        handle.accounts.deinit();
        handle.action_queue.deinit();
        handle.notif_account_map.deinit();
        if (handle.db) |*db| db.deinit();
        handle.allocator.destroy(handle);
    }
}
```

```zig
export fn towncrier_add_account(tc: ?*anyopaque, acct: ?*const AccountDesc) callconv(.c) c_int {
    if (tc == null or acct == null) return 1;
    const handle: *types.TowncrierHandle = @ptrCast(@alignCast(tc.?));
    const desc = acct.?;
    if (desc.token == null) return 1;

    // Copy all strings — caller retains ownership; core must not retain caller's pointers
    const token_slice = std.mem.span(desc.token.?);
    const token_copy = handle.allocator.dupe(u8, token_slice) catch return 1;

    const base_url_slice = if (desc.base_url) |bu| std.mem.span(bu) else "https://api.github.com";
    const base_url_copy = handle.allocator.dupe(u8, base_url_slice) catch {
        handle.allocator.free(token_copy);
        return 1;
    };

    const service: types.Service = switch (desc.service) {
        0 => .github,
        1 => .gitlab,
        else => {
            handle.allocator.free(token_copy);
            handle.allocator.free(base_url_copy);
            return 1;
        },
    };

    const account = types.Account{
        .id = desc.id,
        .service = service,
        .base_url = base_url_copy,
        .token = token_copy,
        .poll_interval_secs = @max(desc.poll_interval_secs, 60),
        .last_modified = null,
    };

    const acct_state = types.AccountState{ .account = account };
    handle.accounts.append(acct_state) catch {
        handle.allocator.free(token_copy);
        handle.allocator.free(base_url_copy);
        return 1;
    };
    return 0;
}
```

Implement `towncrier_snapshot_get`:
```zig
export fn towncrier_snapshot_get(tc: ?*anyopaque) callconv(.c) ?*anyopaque {
    if (tc == null) return null;
    const handle: *types.TowncrierHandle = @ptrCast(@alignCast(tc.?));

    handle.snapshot_lock.lockShared();
    defer handle.snapshot_lock.unlockShared();

    const src = handle.snapshot orelse return null;

    // Deep copy — caller owns result; no lock held after return (per ARCHITECTURE.md)
    const new_snap = handle.allocator.create(types.TowncrierSnapshot) catch return null;
    var new_buf = std.ArrayList(u8).init(handle.allocator);
    new_buf.appendSlice(src.string_buf.items) catch {
        handle.allocator.destroy(new_snap);
        return null;
    };

    const new_items = handle.allocator.dupe(NotificationC, src.items) catch {
        new_buf.deinit();
        handle.allocator.destroy(new_snap);
        return null;
    };

    // Fix up string pointers to point into new_buf (not the original)
    const offset = @intFromPtr(new_buf.items.ptr) - @intFromPtr(src.string_buf.items.ptr);
    for (new_items) |*item| {
        if (item.repo) |p| item.repo = @ptrFromInt(@intFromPtr(p) + offset);
        if (item.title) |p| item.title = @ptrFromInt(@intFromPtr(p) + offset);
        if (item.url) |p| item.url = @ptrFromInt(@intFromPtr(p) + offset);
    }

    new_snap.* = .{
        .items = new_items,
        .string_buf = new_buf,
        .allocator = handle.allocator,
    };
    return new_snap;
}
```

NOTE: The pointer fixup above (rebase string pointers into new_buf) is required because strings point into string_buf.items. When string_buf is deep-copied to a new ArrayList, the base address changes. The offset approach works ONLY if string_buf.items is a single contiguous allocation — which ArrayList guarantees. If the ArrayList was ever reallocated after storing pointers, this would break; but since we copy the entire buf at once and then patch, it is safe.

```zig
export fn towncrier_mark_read(tc: ?*anyopaque, notif_id: u64) callconv(.c) c_int {
    if (tc == null) return 1;
    const handle: *types.TowncrierHandle = @ptrCast(@alignCast(tc.?));

    // Look up account_id from notif_account_map (per RESEARCH.md Open Question #2)
    handle.snapshot_lock.lockShared();
    const account_id_opt = handle.notif_account_map.get(notif_id);
    // Also find the api_id from snapshot
    var api_id_buf: [64]u8 = undefined;
    var api_id_len: usize = 0;
    if (handle.snapshot) |snap| {
        for (snap.items) |item| {
            if (item.id == notif_id) {
                // We need api_id — not in NotificationC directly; use id as string
                // NOTE: GitHub api_id is stored in NotificationC.id (u64 hash of string id)
                // For the PATCH call, we need the original string api_id.
                // Since NotificationC doesn't store api_id as a string, we must use
                // the numeric id and reconstruct, or add api_id to NotificationC.
                // DECISION: Add api_id to the snapshot OR store a separate map.
                // For Phase 2: use the numeric id formatted as a string for the PATCH URL.
                // GitHub thread IDs are numeric strings — formatting u64 back is correct.
                const written = std.fmt.formatIntBuf(&api_id_buf, item.id, 10, .lower, .{});
                api_id_len = written;
                break;
            }
        }
    }
    handle.snapshot_lock.unlockShared();

    const account_id = account_id_opt orelse return 1;

    const api_id = api_id_buf[0..api_id_len];

    const action = types.Action{
        .mark_read = .{
            .notif_id = notif_id,
            .account_id = account_id,
            .api_id = handle.allocator.dupe(u8, api_id) catch return 1,
        },
    };

    handle.action_mutex.lock();
    defer handle.action_mutex.unlock();
    handle.action_queue.append(action) catch {
        handle.allocator.free(action.mark_read.api_id);
        return 1;
    };
    return 0;
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-15 | Tampering | `towncrier_add_account` — AccountDesc strings | mitigate | All string fields (token, base_url) are copied via `allocator.dupe`; caller's original memory is never retained; freeing caller's string after the call is safe |

| T-02-17 | Denial of Service | `towncrier_snapshot_get` — snapshot deep copy | accept | Single-user local app; no multi-tenant risk; copy is bounded by notification count (typically < 200 items) |
